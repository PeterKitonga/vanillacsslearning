3D Buttons
  1. .gradient-button-1 {
  2.     position: relative;
  3.     z-index: 1;
  4.     /* The inline-block value constraints the element to sit inline with
  5.     *  other elements and allows a width and height to be defined.
  6.     *  More info: https://css-tricks.com/almanac/properties/d/display/
  7.     */
  8.     display: inline-block;
  9.     padding: 20px 40px;
  10.     font-size: 1.4rem;
  11.     /* The border-box value for the box-sizing property adds a constraint to the element to prevent
  12.     *  the padding and border added to fit within the declared width of the element.
  13.     *  More info: https://css-tricks.com/almanac/properties/b/box-sizing/
  14.     */
  15.     box-sizing: border-box;
  16.     background-color: #e7eef1;
  17.     border: solid 10px transparent;
  18.     /* The border-image is a shorthand property that lets you use an image or CSS gradient as the border of an element.
  19.     * More info: https://css-tricks.com/almanac/properties/b/border-image/
  20.     * 
  21.     * The linear-gradient value is a property that is considered to natively generate images in CSS. The property comprises of the angle,
  22.     * direction, color stops(start, stop).
  23.     * More info: https://css-tricks.com/snippets/css/css-linear-gradient/
  24.     *
  25.     */
  26.     border-image: linear-gradient( to top right, orangered, yellow );
  27.     border-image-slice: 1;
  28.     color: orangered;
  29.     transition: all 0.3s ease-in-out;
  30. }
