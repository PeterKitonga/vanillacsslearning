In this lesson, I learnt about styling the webpage for print.

Example #1
  1. /* Media queries (@media) are a CSS technique that provide responsive design of web applications. They are responsible for providing the right experience for the right device.
  2. *  Below media query is actually used to style the webpage components to be printed via 'Ctrl + P'.
  3. *  More info on media queries: https://css-tricks.com/snippets/css/media-queries-for-standard-devices/ and https://www.smashingmagazine.com/2011/11/how-to-set-up-a-print-style-sheet/
  4. */
  5. @media print {
  6.     /* HIDE UNWANTED ELEMENTS */
  7.     .course-files,
  8.     #logo,
  9.     .final-result::before,
  10.     .info::before,
  11.     .final-result::after,
  12.     .info::after,
  13.     .final-result h2,
  14.     .info h2,
  15.     img {
  16.         display: none;
  17.     }
  18.  
  19.     /* GLOBAL TYPOGRAPHY RESET */
  20.     body {
  21.         font-family: Georgia, serif;
  22.         font-size: 16pt; /* pt is used as the unit of measurement instead of px since there are no pixels on paper */
  23.         line-height: 24pt;
  24.         background: none;
  25.         color: #000000;
  26.     }
  27.  
  28.     /* LAYOUT RESET */ 
  29.     .final-result,
  30.     .info,
  31.     section,
  32.     pre,
  33.     footer {
  34.         background: none;
  35.         width: 100%;
  36.         border: none;
  37.         padding: 0;
  38.         margin: 0;
  39.         color: #000000;
  40.     }
  41.  
  42.     section:first-child::before {
  43.         border: none;
  44.     }
  45.  
  46.     .shadow-lg {
  47.         box-shadow: none !important;
  48.     }
  49.  
  50.     /* SPECIFICS */
  51.     h1, h2, h3, h4, h5, h6 {
  52.         color: #000000;
  53.         font-size: 24pt;
  54.         line-height: 30pt;
  55.         margin: 0;
  56.     }
  57.  
  58.     h1 {
  59.         font-size: 50pt;
  60.         margin: 0 0 20pt;
  61.         text-align: center;
  62.         width: 100%;
  63.         transform: rotate(0);
  64.     }
  65.  
  66.     .p-2 .btn {
  67.         display: none;
  68.     }
  69.  
  70.     .p-3 {
  71.         padding: 0 !important;
  72.     }
  73.  
  74.     a:link {
  75.         margin: 0;
  76.         padding: 0;
  77.     }
  78.  
  79.     a::after {
  80.         content: " [" attr(href) "]";
  81.     }
  82.  
  83.     footer a:link,
  84.     footer a:visited {
  85.         color: #000000;
  86.         background: none;
  87.     }
  88. }
